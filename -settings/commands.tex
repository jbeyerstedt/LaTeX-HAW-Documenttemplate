% make something wider than text, e.g. listings
% USAGE: \widetext{-1.25cm}{-1.5cm}{ %\lstinputlisting[]{path} }
%   margin-offet:  left     right
\newenvironment{changemargin}[2]{%
  \begin{list}{}{%
    \setlength{\topsep}{0pt}%
    \setlength{\leftmargin}{#1}%
    \setlength{\rightmargin}{#2}%
    \setlength{\listparindent}{\parindent}%
    \setlength{\itemindent}{\parindent}%
    \setlength{\parsep}{\parskip}%
  }%
  \item[]%
}{\end{list}}
\newcommand{\widetext}[3]{\begin{changemargin}{#1}{#2} #3 \end{changemargin}}
